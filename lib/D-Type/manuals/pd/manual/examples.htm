Example 1: From Design To Code

Step 1: Start with a design and identify which PowerDoc objects are needed



Step 2: Assign values to all object properties and links



Step 3: Position visual objects on the page and assign them a rendering function



Step 4: Convert design specifications to code

int MyPowerDocMake(DT_PDDOC power_doc)
{
    /* PART 1: Defining objects and their properties */ 
    /* 6 PowerDoc Object(s) */ 
    DT_ID_SLONG obj[6];

    obj[0] = pdObjAdd(power_doc, 0, "Surface Definition");
    obj[1] = pdObjAdd(power_doc, 0, "Line Definition");
    obj[2] = pdObjAdd(power_doc, 0, "Circle, Big - Smiley's Head");
    obj[3] = pdObjAdd(power_doc, 0, "Circle, Filled - Smiley's Eye");
    obj[4] = pdObjAdd(power_doc, 0, "Ellipse - Smiley's Nose");
    obj[5] = pdObjAdd(power_doc, 0, "Arc - Smiley's Mouth");

    /* Properties for object 0 */ 
    pdPropAdd(power_doc, obj[0], pdRGBT, (DT_UBYTE*)"50 64 14 00", PD_DEFAULT);

    /* Properties for object 1 */ 
    pdPropAdd(power_doc, obj[1], pdLineThickness, (DT_UBYTE*)"400", PD_DEFAULT);
    pdLinkAdd(power_doc, obj[1], pdStylePtr, obj[0]);

    /* Properties for object 2 */ 
    pdPropAdd(power_doc, obj[2], pdRadiusHor, (DT_UBYTE*)"150", PD_DEFAULT);
    pdPropAdd(power_doc, obj[2], pdRadiusVer, (DT_UBYTE*)"150", PD_DEFAULT);
    pdLinkAdd(power_doc, obj[2], pdOutlinePtr, obj[1]);

    /* Properties for object 3 */ 
    pdPropAdd(power_doc, obj[3], pdRadiusHor, (DT_UBYTE*)"15", PD_DEFAULT);
    pdPropAdd(power_doc, obj[3], pdRadiusVer, (DT_UBYTE*)"15", PD_DEFAULT);
    pdLinkAdd(power_doc, obj[3], pdBodyPtr, obj[0]);

    /* Properties for object 4 */ 
    pdPropAdd(power_doc, obj[4], pdRadiusHor, (DT_UBYTE*)"30", PD_DEFAULT);
    pdPropAdd(power_doc, obj[4], pdRadiusVer, (DT_UBYTE*)"15", PD_DEFAULT);
    pdLinkAdd(power_doc, obj[4], pdOutlinePtr, obj[1]);

    /* Properties for object 5 */ 
    pdPropAdd(power_doc, obj[5], pdRadiusHor, (DT_UBYTE*)"90", PD_DEFAULT);
    pdPropAdd(power_doc, obj[5], pdRadiusVer, (DT_UBYTE*)"90", PD_DEFAULT);
    pdPropAdd(power_doc, obj[5], pdAngleAlpha, (DT_UBYTE*)"30", PD_DEFAULT);
    pdPropAdd(power_doc, obj[5], pdAngleTheta, (DT_UBYTE*)"120", PD_DEFAULT);
    pdLinkAdd(power_doc, obj[5], pdLinePtr, obj[1]);


    /* PART 2: Image placement */ 
    /* 5 PowerDoc Image(s) */ 
    pdImgAdd(power_doc, obj[2], 0, 250, 350, 0, 0, 0, pdDrawTrueEllipse, DV_NULL);
    pdImgAdd(power_doc, obj[3], 0, 200, 300, 0, 0, 0, pdDrawTrueEllipse, DV_NULL);
    pdImgAdd(power_doc, obj[3], 0, 300, 300, 0, 0, 0, pdDrawTrueEllipse, DV_NULL);
    pdImgAdd(power_doc, obj[4], 0, 250, 350, 0, 0, 0, pdDrawTrueEllipse, DV_NULL);
    pdImgAdd(power_doc, obj[5], 0, 250, 370, 0, 0, 0, pdDrawArc, DV_NULL);

    return 1;
}
